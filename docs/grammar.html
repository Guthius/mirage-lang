Grammar Reference

Complete EBNF grammar derived from the Mirage parser (src/compiler/ast.cpp). Terminals appear in 'single quotes' or as UPPERCASE token classes. [...] = optional, {...} = zero or more, | = alternative.

Top-Level Structure

program       ::= { declaration } EOF

declaration   ::= [ 'pub' ] fn_decl
               | [ 'pub' ] ext_fn_decl
               | [ 'pub' ] type_decl
               | [ 'pub' ] var_decl
               | [ 'pub' ] macro_decl
               | impl_decl           -- impl cannot be pub

Declarations

Function Declaration

fn_decl       ::= 'fn' IDENT '(' [ param { ',' param } ] ')' [ return_types ] stmt

param         ::= [ 'mut' ] IDENT ':' type

return_types  ::= '->' type
               | '->' '(' type { ',' type } ')'

Extern Function Declaration

ext_fn_decl   ::= 'ext' 'fn' IDENT '(' [ ext_params ] ')' [ '->' type ]

ext_params    ::= ext_param { ',' ext_param } [ ',' '...' ]

ext_param     ::= IDENT ':' type

-- 'ext' is lexed as an identifier, not a keyword

Type Declaration

type_decl     ::= 'type' IDENT '=' type

Variable / Constant Declaration

var_decl      ::= ( 'mut' | 'const' ) IDENT
                  ( ':' type [ '=' expr ]
                  | ':=' expr )

-- 'const' requires an initializer
-- top-level 'const' initializer may be import(...)

Macro Declaration

macro_decl    ::= 'macro' IDENT '(' [ macro_param { ',' macro_param } ] ')' '->' expr

macro_param   ::= IDENT ':' type

Impl Block

impl_decl     ::= 'impl' named_type '{' { method_decl } '}'

method_decl   ::= [ 'pub' ] 'fn' IDENT
                  '(' ( 'self' | 'mut' 'self' )
                      { ',' IDENT ':' type }
                  ')'
                  [ return_types ]
                  stmt

Types

type          ::= '*' type                              -- pointer
               | '[' ']' type                          -- slice
               | '[' expr ']' type                     -- array
               | struct_type
               | enum_type
               | union_type
               | fn_type
               | named_type
               | builtin_type

struct_type   ::= 'struct' [ '(' 'packed' ')' ] '{'
                    { IDENT ':' type [ '=' expr ] }   -- newline-separated fields
                  '}'

enum_type     ::= 'enum' [ '(' type ')' ] '{'
                    { IDENT [ '=' expr ] }             -- newline-separated variants
                  '}'

union_type    ::= 'union' '{'                          -- untagged
                    { IDENT ':' type }
                  '}'
               | 'union' '(' 'enum' ')' '{'            -- tagged
                    { IDENT [ ':' type ] }
                  '}'

fn_type       ::= 'fn' '(' [ fn_type_params ] ')'
                    [ '->' type | '->' '(' type { ',' type } ')' ]

fn_type_params ::= type { ',' type } [ ',' '...' ]

named_type    ::= IDENT { '.' IDENT }

builtin_type  ::= 'u8' | 'u16' | 'u32' | 'u64'
               | 'i8' | 'i16' | 'i32' | 'i64'
               | 'f32' | 'f64'
               | 'usize' | 'bool' | 'byte' | 'error' | 'anyptr'

Statements

stmt          ::= block_stmt
               | if_stmt
               | while_stmt
               | switch_stmt
               | var_decl_stmt
               | continue_stmt
               | break_stmt
               | return_stmt
               | defer_stmt
               | expr_stmt

block_stmt    ::= '{' { stmt } '}'

if_stmt       ::= 'if' expr stmt [ 'else' stmt ]

while_stmt    ::= 'while' expr block_stmt

switch_stmt   ::= 'switch' expr '{' [ switch_arm { ',' switch_arm } [ ',' ] ] '}'

switch_arm    ::= arm_pattern ':' stmt

var_decl_stmt ::= ( 'mut' | 'const' ) IDENT
                  ( ':' type [ '=' expr ]
                  | ',' IDENT { ',' IDENT } ':=' expr   -- group decl
                  | ':=' expr )

continue_stmt ::= 'continue'
break_stmt    ::= 'break'
return_stmt   ::= 'return' [ expr { ',' expr } ]
defer_stmt    ::= 'defer' stmt
expr_stmt     ::= expr

-- In group decl, any name may be '_' (identifier) to discard that slot

Expressions — Precedence (low → high)

expr          ::= assign_expr | import_expr

import_expr   ::= 'import' '(' STRING ')'
                  -- only valid as initializer of const := with no type annotation

assign_expr   ::= ternary_expr [ assign_op assign_expr ]   -- right-associative

assign_op     ::= '=' | '+=' | '-=' | '*=' | '/='
               | '&=' | '|=' | '^=' | '<<=' | '>>='

ternary_expr  ::= logical_or_expr [ '?' expr ':' expr ]   -- right-assoc

logical_or_expr  ::= logical_and_expr { '||' logical_and_expr }

logical_and_expr ::= bitwise_or_expr  { '&&' bitwise_or_expr }

bitwise_or_expr  ::= bitwise_xor_expr { '|'  bitwise_xor_expr }

bitwise_xor_expr ::= bitwise_and_expr { '^'  bitwise_and_expr }

bitwise_and_expr ::= equality_expr    { '&'  equality_expr }

equality_expr    ::= comparison_expr  { ( '==' | '!=' ) comparison_expr }

comparison_expr  ::= shift_expr  { ( '<' | '>' | '<=' | '>=' ) shift_expr }

shift_expr    ::= additive_expr { ( '<<' | '>>' ) additive_expr }

additive_expr ::= mult_expr     { ( '+' | '-' ) mult_expr }

mult_expr     ::= unary_expr    { ( '*' | '/' | '%' ) unary_expr }

unary_expr    ::= 'try' postfix_expr
               | ( '-' | '!' | '~' | '&' | '*' ) unary_expr
               | postfix_expr

-- 'try' has tighter binding than binary operators

postfix_expr  ::= primary_expr { postfix_op }

postfix_op    ::= '(' [ expr { ',' expr } ] ')'            -- call
               | '.' IDENT                                  -- member
               | '.' IDENT '{.' field_init { ',' field_init } '}'  -- qualified tagged variant
               | '[' expr ']'                               -- index
               | '[' expr '..' expr ']'                     -- slice range
               | '++'                                       -- post-increment
               | '--'                                       -- post-decrement

Primary Expressions

primary_expr  ::= INT_LITERAL
               | FLOAT_LITERAL
               | STRING_LITERAL
               | 'true'
               | 'false'
               | 'nil'
               | 'iota'
               | 'default'
               | 'undefined'
               | IDENT
               | '(' expr ')'
               | sizeof_expr
               | len_expr
               | cast_expr
               | match_expr
               | braced_initializer
               | dot_ident_expr
               | contextual_tagged_variant

sizeof_expr   ::= 'sizeof' expr

len_expr      ::= 'len' expr

cast_expr     ::= 'cast' '(' expr ',' type [ ',' expr ] ')'

match_expr    ::= 'match' expr '{' [ match_arm { ',' match_arm } [ ',' ] ] '}'

match_arm     ::= arm_pattern ':' expr

arm_pattern   ::= '.' IDENT [ '(' [ '&' ] IDENT ')' ]   -- variant (enum/tagged-union)
               | '_'                                       -- default / wildcard (must be last)
               | expr                                      -- literal constant (integer or bool)

dot_ident_expr              ::= '.' IDENT
                                -- enum literal; requires expected type context

contextual_tagged_variant   ::= '.' IDENT '{.' field_init { ',' field_init } '}'
                                -- '{.' disambiguates from block statements

field_init    ::= '.' IDENT '=' expr

braced_initializer ::= '{' '}'                                    -- empty (full default)
               | '{' IDENT '=' expr { ',' IDENT '=' expr } '}'   -- struct fields
               | '{' expr { ',' expr } '}'                        -- array values
               -- Disambiguation: '{' IDENT '=' detects struct form; else array form

Literals

INT_LITERAL   ::= decimal_int | hex_int | bin_int

decimal_int   ::= DIGIT { DIGIT | '_' }
hex_int       ::= ( '0x' | '0X' ) HEX_DIGIT { HEX_DIGIT | '_' }
bin_int       ::= ( '0b' | '0B' ) BIN_DIGIT { BIN_DIGIT | '_' }

FLOAT_LITERAL ::= DIGIT { DIGIT } '.' DIGIT { DIGIT }

STRING_LITERAL ::= '"' { char | escape_seq } '"'

escape_seq    ::= '\\' | '\"' | '\n' | '\t' | '\r' | '\0'

IDENT         ::= LETTER { LETTER | DIGIT | '_' }

DIGIT         ::= '0' .. '9'
HEX_DIGIT     ::= '0' .. '9' | 'a' .. 'f' | 'A' .. 'F'
BIN_DIGIT     ::= '0' | '1'
LETTER        ::= 'a' .. 'z' | 'A' .. 'Z' | '_'

Syntax Notes

Array braced initializer {1, 2, 3}Comma

Function parametersComma
Multiple return values in returnComma

Tagged Variant Disambiguation

The sequence {. (left-brace immediately followed by dot) signals a tagged union payload constructor. Any other content after { is a braced initializer or block. This disambiguation applies both in postfix position (Type.variant{.) and in primary position (.variant{.).

Multi-Return Syntax

Multi-return type annotations require parentheses: -> (T1, T2). Multi-return return statements use commas: return a, b. Multi-return calls use group declarations: mut a, b := call().

try Precedence

try is parsed at the unary level, tighter than binary operators. try f(x) + g() parses as (try f(x)) + g(). Use parentheses to chain member access: (try f(x)).field.

Reserved Keywords

break byte cast const continue default defer else enum error false fn for if impl import iota len macro match mut nil return sizeof struct switch true try type undefined union while

Note: for is reserved but not yet implemented. ext is scanned as an identifier.
